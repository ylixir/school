    m\frac{\mathrm{d}^2x}{\mathrm{d}t^2}+kx&=0\\
    x(0)&=\alpha\\
    x'(0)&=\beta
\end{align*}
\subsection*{solution}
``English'' units
\begin{align*}
    s&=3\text{in}\\
    &=\frac{3}{12}\text{ft}=\frac{1}{4}\text{ft}\\
    F&=ks\\
    5&=k\frac{1}{4}\\
    k&=20\text{lb/ft}\\
    F&=mg\\
    5&=m32\\
    m&=\frac{5}{32}\text{slug}
\end{align*}
